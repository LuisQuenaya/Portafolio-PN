Tarea 4: Método de Bisección

Ver PDF 
def f(x):
    return x**3 - x - 2

a, b = 1, 2
for _ in range(10):
    c = (a + b)/2
    if f(a)*f(c) < 0:
        b = c
    else:
        a = c
print("Raíz aproximada:", c)
